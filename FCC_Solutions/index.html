Divisible Digit Sums 
C: 
#include <stdio.h>
 
int main() 
{
    int N, sum = 0;
    scanf("%d", &N);

    int n = N;

    // Calculate sum of digits
    while (n > 0) 
    {
        sum += n % 10;
        n /= 10;
    }

    // Check if S(N) divides N
    if (N % sum == 0) 
    {
        printf("Yes\n");
    } 
    else 
    {
        printf("No\n");
    }
    
    return 0;
}

Majority 
C: 
#include <stdio.h>
int main() 
{
    int n, i, count_for = 0, count_against = 0;
    char opinion[7];
    
    scanf("%d", &n);
    
    for (i = 0; i < n; i++) 
    {
        scanf("%s", opinion);
        if (opinion[0] == 'F') 
        {
            count_for++;
        }
        else 
        {
            count_against++;
        }
    }
    
    if (count_for > count_against) 
    {
        printf("Yes\n");
    }
    else 
    {
        printf("No\n");
    }
    
    return 0;
}

    {

    {

Count Distinct Integers 
C: 
#include <stdio.h>

int main() {
    int n, i, num;
    int freq[1000001] = {0}; // initialize frequency array with all zeros
    
    scanf("%d", &n);
    
    for (i = 0; i < n; i++) {
        scanf("%d", &num);
        freq[num]++; // increment the frequency of the number
    }
    
    int count = 0;
    for (i = 0; i < 1000001; i++) {
        if (freq[i] > 0) {
            count++;
        }
    }
    
    printf("%d\n", count);
    
    return 0;
}

Reversing Substrings 
C: 
#include <stdio.h>
#include <string.h>

int main()
{
    int l, r;
    char s[100001];

    scanf("%d %d %s", &l, &r, s);

    int n = strlen(s);

    for (int i = 0; i < n; i++)
    {
        if (i >= l - 1 && i <= r - 1)
        {
            // reverse the substring from l to r
            printf("%c", s[l + r - i - 2]);
        }
        else
        {
            printf("%c", s[i]);
        }
    }

    printf("\n");

    return 0;
}

Maximizing Digit Sums 
C: 
#include <stdio.h>
int main()
{
    long long int N;
    scanf("%lld", &N);
    
    N++;
    int x = 0;
    
    while (N >= 10)
    {
        N /= 10;
        x++;
    }

    printf("%lld", 9*x + N - 1);
    
    return 0;
}
